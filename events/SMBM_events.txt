# Trigger start of game events
# Delay them to a few days after start so that we don't overwhelm players
event = {
	id = smbm.1
    hide_window = yes

	is_triggered_only = yes
    
    immediate = {
        every_playable_country = {
            limit = {
                is_ai = no
            }
            country_event = {
                id = smbm.2
                days = 10
            }
            country_event = {
                id = smbm.3
                days = 10
            }
            country_event = {
                id = smbm.4
                days = 10
            }
            country_event = {
                id = smbm.6
                days = 30
            }
            country_event = {
                id = smbm.8
                days = 30
            }
        }
    }
}

# Live long and prosper part 1: Common grounds boost
country_event = {
	id = smbm.6
	title = smbm.6.name
	desc = smbm.6.desc
	
	picture = GFX_evt_crisis_defeated
	show_sound = event_celebration
	
	is_triggered_only = yes
	
	trigger = {
        is_ai = no
		has_origin = origin_common_ground
	}

	immediate = {
		# science ship
		create_fleet = {
			effect = {
				set_owner = root.owner

				create_ship = {
					name = "Longevity"
					random_existing_design = science
				}

				set_fleet_stance = evasive
				set_location = root.capital_scope.star

				owner = {
					create_leader = {
						class = scientist
						sub_type = survey
						name = random
						species = owner_main_species
					}
				}
				set_leader = last_created_leader
			}
		}

		owner = {
			country_event = {
				id = smbm.7 days = 182
			}
		}
	}

	option = {
		name = smbm.6.option
		add_resource = {
			influence = 125
		}
	}
}

# Trigger periodically
event = {
	id = smbm.9
    hide_window = yes

	is_triggered_only = yes
    
    immediate = {
        every_playable_country = {
            limit = {
                is_ai = no
                NOT = {
                    # Avoid spamming players who already have it
                    has_modifier = economic_chaos
                }
            }
            country_event = {
                id = smbm.10
            }
        }
    }
}

# Negative economy event
country_event = {
	id = smbm.10
	title = smbm.10.name
	desc = smbm.10.desc
	
	picture = GFX_evt_burning_city
	show_sound = event_burning_city
	
	is_triggered_only = yes
	
	trigger = {
        is_ai = no
        NOT = {
            has_modifier = economic_chaos
        }
        OR = {
            AND = {
                resource_stockpile_compare = {
                    resource = minerals
                    value <= 0
                }
                has_monthly_income = {
                    resource = minerals
                    value < 0
                }
            }
            AND = {
                resource_stockpile_compare = {
                    resource = energy
                    value <= 0
                }
                has_monthly_income = {
                    resource = energy
                    value < 0
                }
            }
            AND = {
                resource_stockpile_compare = {
                    resource = food
                    value <= 0
                }
                has_monthly_income = {
                    resource = food
                    value < 0
                }
            }
        }
	}
    immediate = {
        add_modifier = {
            modifier = economic_chaos
            days = 60
        }
    }

	option = {
		name = smbm.10.option
	}
}
