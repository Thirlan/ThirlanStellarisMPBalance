# Planet event that constantly checks if Crime Lord Deal is active and if so
# Randomly decides if this should cause a crime wave.
planet_event = {
	id = smbm.2
	hide_window = yes
	
	mean_time_to_happen = {
		months = 12
	}
	
	# Optimize checks to only focus on players
	trigger = {
		is_ai = no
		has_modifier = crime_lord_deal
	}
	
	immediate {
		random_list = {
			2 = {
				modifier = {
					factor = 0
					has_modifier = criminal_underworld
				}
				planet_event = { id = smbm.3 }
			}
			4 = {
				modifier = {
					factor = 0
					has_modifier = crime_wave
				}
				planet_event = { id = smbm.4 }
			}
			94 = {
				# Do nothing
			}
		}
	}
}

planet_event = {
	id = smbm.3
	title = smbm.3.name
	desc = smbm.3.desc
	
	picture = GFX_evt_dark_alley
	show_sound = event_criminal_activity
	
	is_triggered_only = yes
	
	immediate = {
		FROM = { planet_event = { id = crime.40 days = 30 } }
		remove_modifier = crime_lord_deal
		add_modifier = { modifier = "crime_lord_deal_broken" days = 1800 }
	}
	
	option = {
		name = smbm.3.option
	}
}
