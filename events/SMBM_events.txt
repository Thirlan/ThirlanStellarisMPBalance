namespace = smbm

# Trigger start of game events
# Delay them to a few days after start so that we don't overwhelm players
event = {
	id = smbm.1
    hide_window = yes

	is_triggered_only = yes
    
    immediate = {
        every_playable_country = {
            limit = {
                is_ai = no
            }
            country_event = {
                id = smbm.2
                days = 5
            }
            country_event = {
                id = smbm.4
                days = 10
            }
        }
    }
}

# Anti-rushing event
country_event = {
	id = smbm.2
	title = smbm.2.name
	desc = smbm.2.desc
	
	picture = GFX_evt_space_station
	show_sound = event_default
	
	is_triggered_only = yes
	
	trigger = {
		is_ai = no
	}
	
	immediate = {
		modifier = {
			country_commander_cap_add = -2
			country_official_cap_add = -2
			country_scientist_cap_add = -1
		}
		add_modifier = {
			modifier = smbm_be_careful_out_there
			days = 7300
		}
	}

	option = {
		name = smbm.2.option
	}
}

# Unholy matrimony: habitability restricted empire combo breaker
country_event = {
	id = smbm.4
	title = smbm.4.name
	desc = smbm.4.desc
	
	picture = GFX_evt_big_landing_ship
	show_sound = event_default
	
	is_triggered_only = yes
	
	trigger = {
		is_ai = no
		OR = {
			AND = {
				is_guided_sapience_empire = yes
				OR = {
					has_origin = origin_shattered_ring
					is_void_dweller_empire = yes
					has_origin = origin_life_seeded
					has_origin = origin_ocean_paradise
				}
			}
			AND = {
				is_robot_empire = yes
				OR = {
					has_origin = origin_shattered_ring
					has_origin = origin_life_seeded
					has_origin = origin_ocean_paradise
				}
			}
		}
	}
	
	immediate = {
		add_modifier = {
			modifier = smbm_different_worlds
			days = -1
		}
	}

	option = {
		name = smbm.4.option
	}
}
