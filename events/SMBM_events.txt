namespace = smbm

# Trigger start of game events
# Delay them to a few days after start so that we don't overwhelm players
event = {
	id = smbm.1
    hide_window = yes

	is_triggered_only = yes
    
    immediate = {
        every_playable_country = {
            limit = {
                is_ai = no
            }
            country_event = {
                id = smbm.6
                days = 1
            }
            country_event = {
                id = smbm.2
                days = 5
            }
            country_event = {
                id = smbm.3
                days = 10
            }
            country_event = {
                id = smbm.4
                days = 10
            }
            country_event = {
                id = smbm.5
                days = 10
            }
            country_event = {
                id = smbm.8
                days = 30
            }
        }
    }
}
